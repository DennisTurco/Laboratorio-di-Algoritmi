217  int heap_remove_min()
218  {
219 
220  if (heap_size <= 0)
221  { // heap vuoto!
222  printf("Errore: heap vuoto\n");
223  return -1;
224  }
225 
226  int &temp_heapMinimo = heap[0];
227  int &temp_heap_min_idx = heap_index[0];
228  int &temp_heapSize = heap[heap_size - 1];
229  int &temp_heap_2 = heap_index[heap_size - 1];
230 
231  int minimo = temp_heapMinimo; // il minimo e' il primo elemento dell'array
232 
233  if (details)
234  printf("Minimo identificato %d\n", minimo);
235 
236  // scambio la radice con l'ultima foglia a destra
237  // il minimo e' stato spostato in fondo --> pronto per l'eliminazione
238  swap(temp_heapMinimo, temp_heapSize);
239  swap(temp_heap_min_idx, temp_heap_2);
240 
241  n_operazione++;
242 
243  // elimino il minimo (ora in fondo all'array)
244  heap_size--;
245 
246  // nella radice c'e' un valore grande (massimo?) -> si se disposto in ordine decrescente
247  int i = 0; // indice di lavoro (parto dalla root)
248 
249  while (i >= 0 && !is_leaf(i))
250  { // garantisco di fermarmi alla foglia
251 
252  int &temp_heap = heap[i]; // temp
253  int &temp_heap_i = heap_index[i];
254 
255  if (details)
256  printf("Lavoro con il nodo in posizione i = %d, valore %d\n", i, temp_heap);
257 
258  int con_chi_mi_scambio = -1; // -1 === non mi scambio con nessuno
259 
260  int child_L = child_L_idx(i);
261  int child_R = child_R_idx(i);
262 
263  int &temp_childL = heap[child_L]; // temp
264 
265  // max L R --> indice
266  if (temp_childL < temp_heap)
267  { // confronto i con L
268  con_chi_mi_scambio = child_L;
269  }
270 
271  // considero figlio destro
272  if (child_R >= 0)
273  { // esiste R
274 
275  int &temp_childR = heap[child_R];
276 
277  // controllare L <-> R e i <-> R
278  if (temp_childR < temp_childL && temp_childR < temp_heap)
279  {
280  con_chi_mi_scambio = child_R;
281  }
282  }
283 
284  if (con_chi_mi_scambio >= 0)
285  {
286 
287  int &temp_heapScambio = heap[con_chi_mi_scambio]; // questo passaggio in realta' non mi fa risparmiare letture perche' usato una sola volta e quando passato a funzione viene gia' trattato come &
288  int &temp_heap_si = heap_index[con_chi_mi_scambio];
289 
290  // swap tra i e con_chi_mi_scambio
291  swap(temp_heap, temp_heapScambio);
292  swap(temp_heap_i, temp_heap_si);
293  }
294

296  }

552  void shortest_path(int n)
553  {
554  V_dist[n] = 0;
555  decrease_key(n, 0);
556  int q_size = n_nodi; // contatore elementi presenti ancora sull'array V_visistato[]
557  while (q_size > 0)
558  {
559  // trova il minimo in coda // V_visitato[n] == 0 && V_dist[n] < best_dist){ // solo se in coda e migliore del nodo corrente
560  float best_dist = heap[0]; // per come √® strutturato il min heap, la radice √® sempre l'elemento pi√π piccolo
561  int best_idx = heap_index[0]; // di conseguenza heap index conterr√† l'indice del minore
562 
563  // potenziale nodo migliore trovato probabilmente eccetto se tutti i nodi hanno distanza infinita se il grafo √® disconnesso
564  if (best_idx >= 0)
565  {
566  // estrai dalla coda
567  int u = best_idx;
568  heap_remove_min();
569  V_visitato[u] = 1; // modifichiamo il flag, nodo trovato
570  q_size--;
571 
572  // esploro la lista di adiacenza
573  node_t *elem = E[u]->head;
574  while (elem != NULL)
575  {
576  int v = elem->val; // best_idx ------> v
577 
578  if (V_visitato[v] == 0 || V_dist[v] == INFTY)
579  {
580  float alt = V_dist[u] + elem->w; // costo per arrivare al nuovo nodo passando per best_idx dove il percorso √® il minore alt<-dist[best_idx]+graph.edges(u,v)
581  // elem -> val; // valore del nodo che √® raggiunto dall'arco considerato con la lista di adiacenza
582  // elem -> w; // peso dell'arco considerato
583 
584  if (alt < V_dist[v])
585  { // trovata una scorciatoia verso v => informo V_dist
586  V_dist[v] = alt;
587  V_prev[v] = u;
588  decrease_key(heap_index[v], alt);
589  }
590  }
591 
592  elem = elem->next;
593  }
594  }
595  else
596  { // coda non vuota e nodi nn raggiungibili -----> finito
597  q_size = 0;
598  }
599  }
600  graph_print();
601  }
